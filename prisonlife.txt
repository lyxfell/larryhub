local function giveViaTouch(name, cf)
	local h = getHRP()
	if not h then notify("Character not ready!"); return end
	local old = h.CFrame
	tp(cf)
	task.wait(0.8)
	local h2 = getHRP()
	if not h2 then notify("Character lost during teleport!"); tp(old); return end
	local touched = false
	for _, part in ipairs(Workspace:GetDescendants()) do
		if part:IsA("BasePart") and (part.Position - h2.Position).Magnitude < 12 then
			local touch = part:FindFirstChildOfClass("TouchTransmitter")
			if touch then
				pcall(function()
					firetouchinterest(h2, part, 0)
					task.wait(0.1)
					firetouchinterest(h2, part, 1)
				end)
				touched = true
			end
		end
	end
	task.wait(0.3)
	local h3 = getHRP()
	if h3 then tp(old) end
	if touched then notify(name .. " obtained!") else notify(name .. " - Try again") end
end

local function smoothTP(targetCFrame, duration)
	local h = getHRP()
	if not h then return end
	local startCFrame = h.CFrame
	local startTime = tick()
	while tick() - startTime < duration do
		local alpha = (tick() - startTime) / duration
		alpha = alpha < 0.5 and 2 * alpha * alpha or 1 - math.pow(-2 * alpha + 2, 2) / 2
		h.CFrame = startCFrame:Lerp(targetCFrame, alpha)
		RunService.Heartbeat:Wait()
	end
	h.CFrame = targetCFrame
end

local function findAndPickupItem(itemName, searchRadius)
	local h = getHRP()
	if not h then return false end
	local closestItem = nil
	local closestDist = searchRadius or 50
	for _, obj in ipairs(Workspace:GetDescendants()) do
		if obj:IsA("BasePart") or obj:IsA("Model") then
			local objName = obj.Name:lower()
			if objName:find(itemName:lower()) or objName:find("card") or objName:find("keycard") or objName:find("m9") then
				local pos = obj:IsA("BasePart") and obj.Position or (obj:FindFirstChildWhichIsA("BasePart") and obj:FindFirstChildWhichIsA("BasePart").Position)
				if pos then
					local dist = (h.Position - pos).Magnitude
					if dist < closestDist then
						closestDist = dist
						closestItem = obj
					end
				end
			end
		end
	end
	if closestItem then
		local pos = closestItem:IsA("BasePart") and closestItem.Position or closestItem:FindFirstChildWhichIsA("BasePart").Position
		smoothTP(CFrame.new(pos.X, pos.Y, pos.Z), 0.5)
		task.wait(0.3)
		local h2 = getHRP()
		if h2 then
			for _, part in ipairs(Workspace:GetDescendants()) do
				if part:IsA("BasePart") and (part.Position - h2.Position).Magnitude < 15 then
					local touch = part:FindFirstChildOfClass("TouchTransmitter")
					if touch then
						pcall(function()
							firetouchinterest(h2, part, 0)
							task.wait(0.1)
							firetouchinterest(h2, part, 1)
						end)
					end
				end
			end
		end
		return true
	end
	return false
end


-- ==========================================
-- UNDERGROUND WALK TO FUNCTION (Avoid arrest)
-- ==========================================
local function walkTo(targetPos, walkSpeed, underground)
	local h = getHRP()
	if not h then return false end
	local hum = getHumanoid()
	if not hum then return false end

	local startPos = h.Position
	local distance = (targetPos - startPos).Magnitude
	local spd = walkSpeed or 50
	local duration = distance / spd
	local startTime = tick()

	-- Underground offset to avoid being seen (-25 studs below ground)
	local undergroundOffset = underground and -25 or 0
	local startY = startPos.Y + undergroundOffset
	local targetY = targetPos.Y + undergroundOffset

	-- Face target direction
	local lookDir = (targetPos - startPos).Unit
	h.CFrame = CFrame.new(startPos.X, startY, startPos.Z) * CFrame.Angles(0, math.atan2(lookDir.X, lookDir.Z), 0)

	while tick() - startTime < duration do
		local alpha = (tick() - startTime) / duration
		-- Smooth easing
		alpha = alpha < 0.5 and 2 * alpha * alpha or 1 - math.pow(-2 * alpha + 2, 2) / 2
		local newPos = startPos:Lerp(targetPos, alpha)
		local newY = startY + (targetY - startY) * alpha
		h.CFrame = CFrame.new(newPos.X, newY, newPos.Z) * CFrame.Angles(0, math.atan2(lookDir.X, lookDir.Z), 0)
		RunService.Heartbeat:Wait()
	end

	h.CFrame = CFrame.new(targetPos.X, targetY, targetPos.Z)
	return true
end

-- ==========================================
-- SHOOT AT TARGET WITH EQUIPPED GUN
-- ==========================================
local function shootAtTarget(targetHRP, shotCount)
	local c = player.Character
	if not c then return false end
	local myHRP = c:FindFirstChild("HumanoidRootPart")
	if not myHRP then return false end

	-- Find equipped tool
	local tool = nil
	for _, t in ipairs(c:GetChildren()) do
		if t:IsA("Tool") then
			tool = t
			break
		end
	end

	-- If no tool equipped, try to equip from backpack
	if not tool then
		for _, t in ipairs(player.Backpack:GetChildren()) do
			if t:IsA("Tool") and (t.Name:find("Shotgun") or t.Name:find("Remington") or t.Name:find("AK") or t.Name:find("M9") or t.Name:find("MP5") or t.Name:find("M4") or t.Name:find("Gun")) then
				local hum = getHumanoid()
				if hum then
					hum:EquipTool(t)
					tool = t
					task.wait(0.4)
				end
				break
			end
		end
	end

	if not tool then return false end

	-- Face the target
	local targetPos = targetHRP.Position
	local myPos = myHRP.Position
	local lookDir = (targetPos - myPos).Unit
	myHRP.CFrame = CFrame.new(myPos, myPos + Vector3.new(lookDir.X, 0, lookDir.Z))

	task.wait(0.2)

	-- Shoot
	local shots = shotCount or 8
	for i = 1, shots do
		pcall(function()
			tool:Activate()
		end)
		task.wait(0.12)
	end

	return true
end

-- ==========================================
-- FIND NEAREST ALIVE GUARD
-- ==========================================
local function findNearestGuard()
	local myHRP = getHRP()
	if not myHRP then return nil end

	local nearestGuard = nil
	local nearestDist = math.huge

	for _, p in ipairs(Players:GetPlayers()) do
		if p ~= player and p.Team and p.Team.Name == "Guards" and p.Character then
			local tHRP = p.Character:FindFirstChild("HumanoidRootPart")
			local tHum = p.Character:FindFirstChildOfClass("Humanoid")
			if tHRP and tHum and tHum.Health > 0 then
				local dist = (myHRP.Position - tHRP.Position).Magnitude
				if dist < nearestDist then
					nearestDist = dist
					nearestGuard = {hrp = tHRP, humanoid = tHum, player = p}
				end
			end
		end
	end

	return nearestGuard
end

-- ==========================================
-- AUTO KEYCARD (Walk Underground + Shoot Version)
-- ==========================================
local function autoGetKeycard()
	if autoRunning then notify("Auto already running!"); return end
	autoRunning = true
	notify("Auto Keycard started...")

	task.spawn(function()
		local h = getHRP()
		if not h then autoRunning = false; return end
		local oldPos = h.CFrame

		-- Enable noclip for underground walking
		local wasNoclip = noclipEnabled
		if not noclipEnabled then
			toggleNoclip()
		end

		-- Step 1: Walk underground to Police Office
		notify("Step 1: Walking to Police Office (underground)...")
		local policeOfficePos = CFrame.new(826.2, 98.99, 2294.85).Position
		walkTo(policeOfficePos, 60, true)
		task.wait(0.5)

		-- Step 2: Walk underground to Police Shotgun location
		notify("Step 2: Getting Police Shotgun...")
		local policeSGPos = CFrame.new(820.298584, 97.8500061, 2229.39624).Position
		walkTo(policeSGPos, 50, true)
		task.wait(0.5)

		-- Come up to surface to pickup shotgun
		local h2 = getHRP()
		if h2 then
			h2.CFrame = CFrame.new(policeSGPos.X, policeSGPos.Y + 3, policeSGPos.Z)
		end
		task.wait(0.3)

		-- Pickup shotgun via touch
		local touched = false
		for _, part in ipairs(Workspace:GetDescendants()) do
			if part:IsA("BasePart") and (part.Position - policeSGPos).Magnitude < 15 then
				local touch = part:FindFirstChildOfClass("TouchTransmitter")
				if touch then
					pcall(function()
						firetouchinterest(h2, part, 0)
						task.wait(0.1)
						firetouchinterest(h2, part, 1)
					end)
					touched = true
				end
			end
		end

		if touched then
			notify("Police Shotgun obtained!")
		else
			notify("Shotgun pickup failed, trying fallback...")
			givePoliceSG()
		end
		task.wait(1.0)

		-- Step 3: Find nearest guard
		notify("Step 3: Looking for guard...")
		local targetGuard = findNearestGuard()

		if targetGuard then
			-- Step 4: Walk underground near guard, then surface
			notify("Step 4: Approaching guard (underground)...")
			local guardPos = targetGuard.hrp.Position
			local approachPos = guardPos + Vector3.new(6, 0, 6)
			walkTo(approachPos, 55, true)
			task.wait(0.3)

			-- Come up to surface behind guard
			local h3 = getHRP()
			if h3 then
				h3.CFrame = CFrame.new(approachPos.X, approachPos.Y + 5, approachPos.Z)
			end
			task.wait(0.3)

			-- Step 5: Shoot guard with shotgun
			notify("Step 5: Shooting guard...")
			local shotSuccess = shootAtTarget(targetGuard.hrp, 10)

			if not shotSuccess then
				notify("Shooting failed, using melee...")
				for i = 1, 5 do
					pcall(function()
						ReplicatedStorage:WaitForChild("meleeEvent"):FireServer(targetGuard.player, 1, 1)
					end)
					task.wait(0.3)
				end
			end

			task.wait(1.5)

			-- Step 6: Search for dropped keycard
			notify("Step 6: Searching for dropped keycard...")
			task.wait(1.5)
			local found = findAndPickupItem("keycard", 30)
			if found then 
				notify("Keycard picked up!") 
			else 
				giveKeycard() 
			end
		else
			notify("No guard found, trying direct method...")
			giveKeycard()
		end

		-- Step 7: Walk back underground
		notify("Step 7: Returning (underground)...")
		task.wait(0.5)
		walkTo(oldPos.Position, 60, true)

		-- Come up at original position
		local h4 = getHRP()
		if h4 then
			h4.CFrame = oldPos
		end

		-- Restore noclip state
		if not wasNoclip and noclipEnabled then
			toggleNoclip()
		end

		autoRunning = false
		notify("Auto Keycard complete!")
	end)
end

		local h2 = getHRP()
		if h2 then
			h2.CFrame = CFrame.new(policeSGPos.X, policeSGPos.Y + 3, policeSGPos.Z)
		end
		task.wait(0.3)

		local touched = false
		for _, part in ipairs(Workspace:GetDescendants()) do
			if part:IsA("BasePart") and (part.Position - policeSGPos).Magnitude < 15 then
				local touch = part:FindFirstChildOfClass("TouchTransmitter")
				if touch then
					pcall(function()
						firetouchinterest(h2, part, 0)
						task.wait(0.1)
						firetouchinterest(h2, part, 1)
					end)
					touched = true
				end
			end
		end

		if touched then
			notify("Police Shotgun obtained!")
		else
			notify("Shotgun pickup failed, trying fallback...")
			givePoliceSG()
		end
		task.wait(1.0)

		-- Step 3: Find nearest guard
		notify("Step 3: Looking for guard...")
		local targetGuard = findNearestGuard()

		if targetGuard then
			-- Step 4: Walk underground near guard
			notify("Step 4: Approaching guard (underground)...")
			local guardPos = targetGuard.hrp.Position
			local approachPos = guardPos + Vector3.new(6, 0, 6)
			walkTo(approachPos, 55, true)
			task.wait(0.3)

			-- Come up to surface
			local h3 = getHRP()
			if h3 then
				h3.CFrame = CFrame.new(approachPos.X, approachPos.Y + 5, approachPos.Z)
			end
			task.wait(0.3)

			-- Step 5: Shoot guard
			notify("Step 5: Shooting guard...")
			local shotSuccess = shootAtTarget(targetGuard.hrp, 10)

			if not shotSuccess then
				notify("Shooting failed, using melee...")
				for i = 1, 5 do
					pcall(function()
						ReplicatedStorage:WaitForChild("meleeEvent"):FireServer(targetGuard.player, 1, 1)
					end)
					task.wait(0.3)
				end
			end

			task.wait(1.5)

			-- Step 6: Search for dropped M9
			notify("Step 6: Searching for dropped M9...")
			task.wait(1.5)
			local found = findAndPickupItem("m9", 30)
			if found then 
				notify("M9 picked up!") 
			else 
				giveM9() 
			end
		else
			notify("No guard found, trying direct method...")
			giveM9()
		end

		-- Step 7: Walk back underground
		notify("Step 7: Returning (underground)...")
		task.wait(0.5)
		walkTo(oldPos.Position, 60, true)

		-- Come up at original position
		local h4 = getHRP()
		if h4 then
			h4.CFrame = oldPos
		end

		-- Restore noclip state
		if not wasNoclip and noclipEnabled then
			toggleNoclip()
		end

		autoRunning = false
		notify("Auto M9 complete!")
	end)
end

local function toggleFly()
	flyEnabled = not flyEnabled
	if flyEnabled then
		notify("Fly ON (WASD+Space/Shift)")
		local bg = Instance.new("BodyGyro")
		bg.P, bg.MaxTorque = 9e4, Vector3.new(9e9, 9e9, 9e9)
		bg.CFrame = getHRP().CFrame; bg.Parent = getHRP()
		local bv = Instance.new("BodyVelocity")
		bv.Velocity, bv.MaxForce = Vector3.new(0,0,0), Vector3.new(9e9, 9e9, 9e9)
		bv.Parent = getHRP()
		flyConn = RunService.RenderStepped:Connect(function()
			if not flyEnabled then return end
			local h = getHRP(); if not h then return end
			local cam = Workspace.CurrentCamera
			local md = Vector3.new(0,0,0)
			if UserInputService:IsKeyDown(Enum.KeyCode.W) then md = md + cam.CFrame.LookVector end
			if UserInputService:IsKeyDown(Enum.KeyCode.S) then md = md - cam.CFrame.LookVector end
			if UserInputService:IsKeyDown(Enum.KeyCode.A) then md = md - cam.CFrame.RightVector end
			if UserInputService:IsKeyDown(Enum.KeyCode.D) then md = md + cam.CFrame.RightVector end
			if UserInputService:IsKeyDown(Enum.KeyCode.Space) then md = md + Vector3.new(0,1,0) end
			if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then md = md - Vector3.new(0,1,0) end
			if md.Magnitude > 0 then md = md.Unit * 50 * flySpeed end
			bv.Velocity = md; bg.CFrame = cam.CFrame
		end)
	else
		notify("Fly OFF")
		if flyConn then flyConn:Disconnect(); flyConn = nil end
		local h = getHRP()
		if h then for _, c in ipairs(h:GetChildren()) do if c:IsA("BodyGyro") or c:IsA("BodyVelocity") then c:Destroy() end end end
	end

local function toggleJumpHack()
	jumphackEnabled = not jumphackEnabled
	local h = getHumanoid()
	if jumphackEnabled then
		notify("JumpHack ON (x" .. jumpMultiplier .. ")")
		if h then
			-- Try JumpPower first, then JumpHeight as fallback
			pcall(function()
				h.JumpPower = originalJumpPower * jumpMultiplier
			end)
			pcall(function()
				h.JumpHeight = originalJumpPower * jumpMultiplier * 0.5
			end)
		end
		-- Use less frequent updates to avoid anti-cheat
		jumpHackConn = RunService.Heartbeat:Connect(function()
			if not jumphackEnabled then return end
			local hm = getHumanoid()
			if hm then
				pcall(function()
					if hm.JumpPower ~= originalJumpPower * jumpMultiplier then
						hm.JumpPower = originalJumpPower * jumpMultiplier
					end
				end)
			end
		end)
	else
		notify("JumpHack OFF")
		if jumpHackConn then jumpHackConn:Disconnect(); jumpHackConn = nil end
		if h then
			pcall(function() h.JumpPower = originalJumpPower end)
			pcall(function() h.JumpHeight = originalJumpPower * 0.5 end)
		end
	end
end

			lastUpdate = now

			hue = hue + 0.015
			if hue > 1 then hue = 0 end
			local color = Color3.fromHSV(hue, 1, 1)
			local c = player.Character
			if not c then return end

			for _, t in ipairs(c:GetChildren()) do
				if t:IsA("Tool") then
					-- Color all descendants including meshes
					local descendants = t:GetDescendants()
					for i = 1, #descendants do
						local p = descendants[i]
						if p:IsA("BasePart") or p:IsA("MeshPart") then
							pcall(function() p.Color = color end)
							pcall(function() p.BrickColor = BrickColor.new(color) end)
						end
						if p:IsA("Trail") then
							pcall(function() p.Color = ColorSequence.new(color) end)
						end
						if p:IsA("ParticleEmitter") then
							pcall(function() p.Color = ColorSequence.new(color) end)
						end
						if p:IsA("PointLight") or p:IsA("SpotLight") then
							pcall(function() p.Color = color end)
						end
						if p:IsA("SpecialMesh") then
							pcall(function() p.VertexColor = Vector3.new(color.R, color.G, color.B) end)
						end
					end
					-- Color the tool itself if it is a part
					if t:IsA("BasePart") or t:IsA("MeshPart") then
						pcall(function() t.Color = color end)
					end
				end
			end
		end)
	else
		notify("RainbowGun OFF")
		if rainbowConn then rainbowConn:Disconnect(); rainbowConn = nil end
	end
end

local function changeShootColor()
	local cols = {
		Color3.fromRGB(255, 50, 50), Color3.fromRGB(50, 255, 50), Color3.fromRGB(50, 50, 255),
		Color3.fromRGB(255, 255, 50), Color3.fromRGB(255, 50, 255), Color3.fromRGB(50, 255, 255),
		Color3.fromRGB(255, 150, 50), Color3.fromRGB(150, 50, 255)
	}
	local col = cols[math.random(1, #cols)]
	local c = player.Character
	if not c then notify("Equip a gun first!"); return end

	local changed = false
	for _, t in ipairs(c:GetChildren()) do
		if t:IsA("Tool") then
			local descendants = t:GetDescendants()
			for i = 1, #descendants do
				local ch = descendants[i]
				if ch:IsA("BasePart") or ch:IsA("MeshPart") then
					pcall(function() ch.Color = col end)
					pcall(function() ch.BrickColor = BrickColor.new(col) end)
				end
				if ch:IsA("Trail") then
					pcall(function() ch.Color = ColorSequence.new(col) end)
				end
				if ch:IsA("ParticleEmitter") then
					pcall(function() ch.Color = ColorSequence.new(col) end)
				end
				if ch:IsA("PointLight") or ch:IsA("SpotLight") then
					pcall(function() ch.Color = col end)
				end
				if ch:IsA("Beam") then
					pcall(function() ch.Color = ColorSequence.new(col) end)
				end
				if ch:IsA("Decal") or ch:IsA("Texture") then
					pcall(function() ch.Color3 = col end)
				end
				if ch:IsA("SpecialMesh") then
					pcall(function() ch.VertexColor = Vector3.new(col.R, col.G, col.B) end)
				end
			end
			if t:FindFirstChild("Handle") and (t.Handle:IsA("BasePart") or t.Handle:IsA("MeshPart")) then
				pcall(function() t.Handle.Color = col end)
				changed = true
			end
			if t:IsA("BasePart") or t:IsA("MeshPart") then
				pcall(function() t.Color = col end)
				changed = true
			end
		end
	end
	if changed then notify("Shoot color changed!") else notify("Equip a gun first!") end
end

local dragging, dragInput, dragStart, startPos
TitleBar.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true; dragStart = input.Position; startPos = Main.Position
		input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
	end
end)
TitleBar.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)
UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		local delta = input.Position - dragStart
		Main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)

-- ==========================================
-- INSERT TOGGLE
-- ==========================================
local insertConnection = nil

local function setupInsertToggle()
	if insertConnection then
		insertConnection:Disconnect()
		insertConnection = nil
	end

	insertConnection = UserInputService.InputBegan:Connect(function(input, gp)
		if gp then return end
		if input.KeyCode == Enum.KeyCode.Insert then
			if SG.Enabled then
				-- Hide animation
				TweenService:Create(Main, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
					Size = UDim2.new(0, 0, 0, 0),
					Position = UDim2.new(0.5, 0, 0.5, 0),
					BackgroundTransparency = 1
				}):Play()
				TweenService:Create(blur, TweenInfo.new(0.3), {Size = 0}):Play()
				task.delay(0.3, function()
					SG.Enabled = false
				end)
			else
				-- Show animation
				SG.Enabled = true
				Main.Size = UDim2.new(0, 0, 0, 0)
				Main.Position = UDim2.new(0.5, 0, 0.5, 0)
				Main.BackgroundTransparency = 1

				TweenService:Create(Main, TweenInfo.new(0.4, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
					Size = UDim2.new(0, 600, 0, 440),
					Position = UDim2.new(0.5, -300, 0.5, -220),
					BackgroundTransparency = 0
				}):Play()
				TweenService:Create(blur, TweenInfo.new(0.4), {Size = 8}):Play()
			end
		end
	end)
end

-- RAINBOW BORDER ANIMATION

local function makeRainbowBar(parent, pos, size, name)
	local bar = Instance.new("Frame", parent)
	bar.Name = name; bar.BorderSizePixel = 0; bar.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
	bar.Size = size; bar.Position = pos; bar.ZIndex = 10
	return bar
end

local topBar = makeRainbowBar(Main, UDim2.new(0, 0, 0, 0), UDim2.new(1, 0, 0, 3), "TopRainbow")
local bottomBar = makeRainbowBar(Main, UDim2.new(0, 0, 1, -3), UDim2.new(1, 0, 0, 3), "BottomRainbow")
local leftBar = makeRainbowBar(Main, UDim2.new(0, 0, 0, 0), UDim2.new(0, 3, 1, 0), "LeftRainbow")
local rightBar = makeRainbowBar(Main, UDim2.new(1, -3, 0, 0), UDim2.new(0, 3, 1, 0), "RightRainbow")

local rainbowBars = {topBar, bottomBar, leftBar, rightBar}
task.spawn(function()
	local hue = 0
	while SG and SG.Parent do
		task.wait(0.03)
		hue = hue + 0.003
		if hue > 1 then hue = 0 end
		local color = Color3.fromHSV(hue, 1, 1)
		for _, bar in ipairs(rainbowBars) do
			if bar and bar.Parent then bar.BackgroundColor3 = color end
		end
	end
end)

-- FLOATING PARTICLES EFFECT (Subtle background)

local particleContainer = Instance.new("Frame", Main)
particleContainer.Size = UDim2.new(1, 0, 1, 0)
particleContainer.BackgroundTransparency = 1
particleContainer.ZIndex = 0

for i = 1, 8 do
	local p = Instance.new("Frame", particleContainer)
	p.Size = UDim2.new(0, math.random(2, 4), 0, math.random(2, 4))
	p.Position = UDim2.new(math.random(), 0, math.random(), 0)
	p.BackgroundColor3 = Color3.fromRGB(100, 170, 255)
	p.BackgroundTransparency = 0.8
	p.BorderSizePixel = 0
	Instance.new("UICorner", p).CornerRadius = UDim.new(1, 0)
	p.ZIndex = 0

	-- Animate each particle
	task.spawn(function()
		local speed = 0.2 + math.random() * 0.3
		while p and p.Parent do
			local currentY = p.Position.Y.Scale
			local newY = currentY - (speed * 0.01)
			if newY < -0.05 then newY = 1.05 end
			p.Position = UDim2.new(p.Position.X.Scale, 0, newY, 0)
			task.wait(0.05)
		end
	end)
end
